get_features <- function(gene_symbols, seurat_obj) {
  rn <- rownames(seurat_obj)
  matched <- c()
  for (g in gene_symbols) {
    hit <- grep(paste0("\\.", g, "$"), rn, value = TRUE)  
    if (length(hit) > 0) {
      matched <- c(matched, hit[1])  
    } else {
      message("Gene not found: ", g)
    }
  }
  return(matched)
}
